}

class ProjectManager {
    constructor() {
        this.project = initializeProject();
        this.status = "En développement";
    }
    
    start() {
        console.log(`Démarrage du projet: ${this.project.name}`);
        // Code de démarrage du projet
    }
    
    update() {
        console.log("Mise à jour du projet...");
        // Code de mise à jour du projet
    }
    
    getStatus() {
        return this.status;
    }
}

// Initialisation du gestionnaire de projet
const manager = new ProjectManager();
manager.start();

// Exemple d'utilisation
function runProject() {
    try {
        manager.update();
        console.log(`Statut actuel: ${manager.getStatus()}`);
    } catch (error) {
        console.error("Une erreur est survenue:", error);
    }
}

runProject();
