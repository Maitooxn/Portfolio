public List<Point> findGroup(int x, int y) {

        return new ArrayList<>();
    }
    
    int targetColor = grid[y][x].getColorIndex();
    List<Point> group = new ArrayList<>();
    boolean[][] visited = new boolean[height][width];
    
    // Utiliser une file pour la recherche en largeur (BFS)
    Queue<Point> queue = new LinkedList<>();
    queue.add(new Point(x, y));
    visited[y][x] = true;
    
    while (!queue.isEmpty()) {
        Point p = queue.poll();
        group.add(p);
        
        // V√©rifier les quatre directions adjacentes
        int[][] directions = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}}; // Bas, Droite, Haut, Gauche
        
        for (int[] dir : directions) {
            int nx = p.x + dir[0];
            int ny = p.y + dir[1];
            
            if (nx >= 0 && nx < width && ny >= 0 && ny < height && 
                !visited[ny][nx] && grid[ny][nx] != null && 
                grid[ny][nx].getColorIndex() == targetColor) {
                
                queue.add(new Point(nx, ny));

        }
